<script>
    // Obtener todos los elementos necesarios del DOM
    const generateBtn = document.getElementById('generateBtn');
    const resultsSection = document.getElementById('resultsSection');
    const loadingIndicator = document.getElementById('loadingIndicator');
    const outfitText = document.getElementById('outfitText');

    // Añadir un evento de clic al botón de generar
    generateBtn.addEventListener('click', async () => {
        // Mostrar la sección de resultados y el indicador de carga
        resultsSection.classList.remove('hidden');
        loadingIndicator.classList.remove('hidden');
        outfitText.textContent = '';
        
        // Obtener los valores de los menús desplegables
        const gender = document.getElementById('gender').value;
        const skinTone = document.getElementById('skinTone').value;
        const hairColor = document.getElementById('hairColor').value;
        const eyeColor = document.getElementById('eyeColor').value;
        const season = document.getElementById('season').value;

        // Construir el prompt para el modelo generativo
        const prompt = `Actúa como un estilista de moda personal. Basándote en las siguientes características, describe un atuendo apropiado para una persona de género ${gender}, especificando las prendas, los colores y los accesorios. Justifica por qué estos elementos combinan bien con las características de la persona y la estación.

        Características de la persona:
        - Género: ${gender}
        - Tono de piel: ${skinTone}
        - Color de pelo: ${hairColor}
        - Color de ojos: ${eyeColor}
        - Estación: ${season}

        Tu respuesta debe ser una descripción detallada y única del atuendo, presentada de forma atractiva y fácil de leer.`;

        // Configuración del endpoint de la API y la clave
        const apiKey = ""; // La clave de API se inyectará automáticamente en el entorno de ejecución
        const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-05-20:generateContent?key=${apiKey}`;

        let retries = 0;
        const maxRetries = 3;
        const initialDelay = 1000; // 1 segundo

        const callApi = async () => {
            try {
                // Preparar el payload para la llamada a la API
                const payload = {
                    contents: [{
                        parts: [{ text: prompt }]
                    }]
                };

                // Realizar la llamada a la API
                const response = await fetch(apiUrl, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json'
                    },
                    body: JSON.stringify(payload)
                });

                if (!response.ok) {
                    // Implementar reintentos con exponential backoff si se excede el límite de velocidad
                    if (response.status === 429 && retries < maxRetries) {
                        const delay = initialDelay * Math.pow(2, retries);
                        retries++;
                        console.log(`Límite de velocidad excedido. Reintentando en ${delay / 1000} segundos...`);
                        await new Promise(res => setTimeout(res, delay));
                        return await callApi(); // Reintentar la llamada
                    }
                    throw new Error(`Error HTTP! estado: ${response.status}`);
                }

                const result = await response.json();
                
                // Extraer el texto generado de la respuesta
                const text = result?.candidates?.[0]?.content?.parts?.[0]?.text;
                
                if (text) {
                    // Mostrar el texto generado y ocultar el indicador de carga
                    outfitText.innerHTML = text.replace(/\n/g, '<br>');
                } else {
                    outfitText.textContent = "Lo siento, no pude generar una sugerencia de atuendo. Por favor, inténtalo de nuevo.";
                }
            } catch (error) {
                console.error('Error al generar el atuendo:', error);
                outfitText.textContent = "Ocurrió un error al conectar con el servidor. Por favor, revisa tu conexión e inténtalo de nuevo.";
            } finally {
                // Ocultar siempre el indicador de carga al finalizar
                loadingIndicator.classList.add('hidden');
            }
        };

        // Llamada inicial a la API
        await callApi();
    });
</script>
